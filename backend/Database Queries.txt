CREATE DATABASE royal;
USE royal;

CREATE TABLE User (
    userId INTEGER PRIMARY KEY AUTO_INCREMENT,
    firstName VARCHAR(100) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role VARCHAR(50)
);

CREATE TABLE Project (
    projectId INTEGER PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    description VARCHAR(500),
    technology VARCHAR(255),
    estimatedHours INTEGER,
    startDate DATE,
    completionDate DATE
);

CREATE TABLE Project_team (
    project_team_id INTEGER PRIMARY KEY AUTO_INCREMENT,
    projectId INTEGER,
    userId INTEGER,
    FOREIGN KEY (projectId) REFERENCES Project(projectId) ON DELETE CASCADE,
    FOREIGN KEY (userId) REFERENCES User(userId) ON DELETE CASCADE
);

CREATE TABLE Status (
    statusId INTEGER PRIMARY KEY AUTO_INCREMENT,
    statusName VARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE Project_module (
    moduleId INTEGER PRIMARY KEY AUTO_INCREMENT,
    projectId INTEGER,
    moduleName VARCHAR(255) NOT NULL,
    description VARCHAR(500),
    estimatedHours INTEGER,
    status INTEGER,
    startDate DATE,
    FOREIGN KEY (projectId) REFERENCES Project(projectId) ON DELETE CASCADE,
    FOREIGN KEY (status) REFERENCES Status(statusId) ON DELETE SET NULL
);

CREATE TABLE Task (
    taskId INTEGER PRIMARY KEY AUTO_INCREMENT,
    moduleId INTEGER,
    projectId INTEGER,
    title VARCHAR(255) NOT NULL,
    priority VARCHAR(50),
    description VARCHAR(500) NOT NULL,
    statusId INTEGER,
    totalMinutes INTEGER,
    FOREIGN KEY (moduleId) REFERENCES Project_module(moduleId) ON DELETE CASCADE,
    FOREIGN KEY (projectId) REFERENCES Project(projectId) ON DELETE CASCADE,
    FOREIGN KEY (statusId) REFERENCES Status(statusId) ON DELETE SET NULL
);

CREATE TABLE User_task (
    user_task_id INTEGER PRIMARY KEY AUTO_INCREMENT,
    userId INTEGER,
    taskId INTEGER,
    FOREIGN KEY (userId) REFERENCES User(userId) ON DELETE CASCADE,
    FOREIGN KEY (taskId) REFERENCES Task(taskId) ON DELETE CASCADE
);
